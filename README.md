> **Setup:** Add this GitHub Action to `.github/workflows/snake.yml` in your profile repo to generate the snake animation automatically ↓

```yaml
name: Generate Snake

on:
  schedule:
    - cron: "0 */12 * * *"
  workflow_dispatch:
  push:
    branches:
      - main

jobs:
  generate:
    permissions:
      contents: write
    runs-on: ubuntu-latest
    steps:
      - uses: Platane/snk@v3
        with:
          github_user_name: ${{ github.repository_owner }}
          outputs: |
            dist/github-contribution-grid-snake.svg
            dist/github-contribution-grid-snake-dark.svg?palette=github-dark
      - uses: crazy-max/ghaction-github-pages@v3.1.0
        with:
          target_branch: output
          build_dir: dist
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```

function tickPac() {
  if(!pacRunning) return;
  // Try next dir
  const [nx,ny]=[pacman.x+pacNextDir[0], pacman.y+pacNextDir[1]];
  if(nx>=0&&nx<MAZE_COLS&&ny>=0&&ny<MAZE_ROWS&&maze[ny][nx]!==1) pacDir=[...pacNextDir];
  // Move
  const [mx,my]=[pacman.x+pacDir[0], pacman.y+pacDir[1]];
  if(mx>=0&&mx<MAZE_COLS&&my>=0&&my<MAZE_ROWS&&maze[my][mx]!==1){
    pacman.x=mx; pacman.y=my;
  }
  // Eat
  const cell=maze[pacman.y][pacman.x];
  if(cell===0){maze[pacman.y][pacman.x]=2;pacScore+=10;}
  if(cell===3){maze[pacman.y][pacman.x]=2;pacScore+=50;pacPower=150;ghosts.forEach(g=>g.scared=true);}
  document.getElementById('pacScore').textContent=pacScore;
  if(pacScore>pacHiScore){pacHiScore=pacScore;document.getElementById('pacHi').textContent=pacHiScore;}

  // Power countdown
  if(pacPower>0){pacPower--;if(pacPower===0)ghosts.forEach(g=>g.scared=false);}

  // Move ghosts
  ghosts.forEach(g=>{
    const dirs=[[1,0],[-1,0],[0,1],[0,-1]];
    let best=null,bestScore=-Infinity;
    dirs.forEach(d=>{
      const gx=g.x+d[0],gy=g.y+d[1];
      if(gx<0||gx>=MAZE_COLS||gy<0||gy>=MAZE_ROWS||maze[gy][gx]===1) return;
      if(d[0]===-g.dir[0]&&d[1]===-g.dir[1]) return;
      const dist=g.scared ? (Math.random()*10) : -(Math.abs(gx-pacman.x)+Math.abs(gy-pacman.y));
      if(dist>bestScore){bestScore=dist;best=d;}
    });
    if(best){g.dir=best;g.x+=best[0];g.y+=best[1];}
  });

  // Collision
  ghosts.forEach(g=>{
    if(Math.abs(g.x-pacman.x)<=0&&Math.abs(g.y-pacman.y)<=0){
      if(g.scared){g.x=9;g.y=9;g.scared=false;pacScore+=200;}
      else{
        pacLives--;
        document.getElementById('pacLives').textContent=pacLives;
        if(pacLives<=0){pacRunning=false;drawPacGameOver();return;}
        pacman={x:9,y:16,mouth:0.2,mouthDir:1};
        ghosts.forEach((gh,i)=>{gh.x=8+i%2;gh.y=9+Math.floor(i/2);});
      }
    }
  });

  // Win
  if(!maze.some(row=>row.some(c=>c===0||c===3))){
    pacRunning=false;
    drawPacWin();
    return;
  }

  // Mouth
  pacman.mouth+=0.08*pacman.mouthDir;
  if(pacman.mouth>0.3||pacman.mouth<0.02) pacman.mouthDir*=-1;

  drawPac();
}

function newTetPiece(p){ return {shape:p.shape.map(r=>[...r]),color:p.color,x:Math.floor(TW/2)-Math.floor(p.shape[0].length/2),y:0}; }
function randPiece(){ return PIECES[Math.floor(Math.random()*PIECES.length)]; }

function startTetris(){

function tetValid(p,dx=0,dy=0,shape=null){
  const s=shape||p.shape;
  return s.every((row,r)=>row.every((c,col)=>!c||
    (p.x+col+dx>=0&&p.x+col+dx<TW&&p.y+r+dy<TH&&
    (p.y+r+dy<0||!tetBoard[p.y+r+dy][p.x+col+dx]))));
}

function tetPlace(){
  tetPiece.shape.forEach((row,r)=>row.forEach((c,col)=>{
    if(c&&tetPiece.y+r>=0) tetBoard[tetPiece.y+r][tetPiece.x+col]=tetPiece.color;
  }));
  // Clear lines
  let cleared=0;
  for(let r=TH-1;r>=0;r--){
    if(tetBoard[r].every(c=>c)){
      tetBoard.splice(r,1);
      tetBoard.unshift(Array(TW).fill(0));
      cleared++;r++;
    }
  }
  if(cleared){
    const pts=[0,100,300,500,800][cleared]*tetLevel;
    tetScore+=pts;tetLines+=cleared;
    tetLevel=Math.floor(tetLines/10)+1;
    document.getElementById('tetScore').textContent=tetScore;
    document.getElementById('tetLines').textContent=tetLines;
    document.getElementById('tetLevel').textContent=tetLevel;
    clearInterval(tetInterval);
    tetInterval=setInterval(tetTick,Math.max(80,500-tetLevel*40));
  }
  tetPiece=newTetPiece(tetNext);
  tetNext=randPiece();
  drawTetNext();
  if(!tetValid(tetPiece)){tetRunning=false;drawTetGameOver();}
}

function tetTick(){ if(!tetRunning)return; if(tetValid(tetPiece,0,1)){tetPiece.y++;}else{tetPlace();} drawTet(); }
function tetMoveLeft(){  if(tetRunning&&tetValid(tetPiece,-1,0)){tetPiece.x--;drawTet();} }
function tetMoveRight(){ if(tetRunning&&tetValid(tetPiece,1,0)){tetPiece.x++;drawTet();} }
function tetMoveDown(){  if(tetRunning){if(tetValid(tetPiece,0,1)){tetPiece.y++;}else{tetPlace();}drawTet();} }
function tetRotate(){
  if(!tetRunning)return;
  const rot=tetPiece.shape[0].map((_,i)=>tetPiece.shape.map(r=>r[i]).reverse());
  if(tetValid(tetPiece,0,0,rot)){tetPiece.shape=rot;drawTet();}
  else if(tetValid(tetPiece,1,0,rot)){tetPiece.shape=rot;tetPiece.x++;drawTet();}
  else if(tetValid(tetPiece,-1,0,rot)){tetPiece.shape=rot;tetPiece.x--;drawTet();}
}
function tetHardDrop(){ if(!tetRunning)return; while(tetValid(tetPiece,0,1)){tetPiece.y++;} tetPlace();drawTet(); }

  for(let c=0;c<=TW;c++){tx.beginPath();tx.moveTo(c*TS,0);tx.lineTo(c*TS,tc.height);tx.stroke();}
  // Board
  tetBoard.forEach((row,r)=>row.forEach((c,col)=>{
    if(c){tx.fillStyle=c;tx.shadowColor=c;tx.shadowBlur=6;tx.fillRect(col*TS+1,r*TS+1,TS-2,TS-2);tx.shadowBlur=0;}
  }));
  // Ghost
  if(tetRunning){
    let ghost={...tetPiece,shape:tetPiece.shape.map(r=>[...r])};
    while(tetValid(ghost,0,1)){ghost.y++;}
    ghost.shape.forEach((row,r)=>row.forEach((c,col)=>{
      if(c){tx.fillStyle='rgba(255,255,255,0.08)';tx.strokeStyle='rgba(255,255,255,0.2)';
        tx.fillRect((ghost.x+col)*TS+1,(ghost.y+r)*TS+1,TS-2,TS-2);

    }));

/* ═══════════════════════════════════════════
   🐦 FLAPPY BIRD
═══════════════════════════════════════════ */
const fc = document.getElementById('flappyCanvas');
const fx = fc.getContext('2d');
const FW=320,FH=480,PIPE_W=52,GAP=130,PIPE_SPEED=2.5;
let bird,pipes,flappyScore,flappyBest=0,flappyRunning=false,flappyFrame,flappyState='idle';

function flappyFlap(){ if(flappyState==='idle'||flappyState==='over'){startFlappy();return;} bird.vy=-7; }

function startFlappy(){
  bird={x:80,y:FH/2,vy:0,radius:14,rot:0};
  pipes=[];flappyScore=0;flappyState='running';flappyRunning=true;
  document.getElementById('flappyScore').textContent='0';
  cancelAnimationFrame(flappyFrame);
  flappyLoop();
}

function flappyLoop(){
  if(!flappyRunning){drawFlappyOver();return;}
  tickFlappy();drawFlappy();
  flappyFrame=requestAnimationFrame(flappyLoop);
}

function tickFlappy(){
  bird.vy+=0.38; bird.y+=bird.vy;
  bird.rot=Math.max(-25,Math.min(90,bird.vy*4));
  // Pipes
  if(!pipes.length||pipes[pipes.length-1].x<FW-180){
    const top=60+Math.random()*(FH-GAP-120);
    pipes.push({x:FW,top,scored:false});
  }
  pipes.forEach(p=>p.x-=PIPE_SPEED);
  pipes=pipes.filter(p=>p.x>-PIPE_W);
  // Score
  pipes.forEach(p=>{
    if(!p.scored&&p.x+PIPE_W<bird.x){
      p.scored=true;flappyScore++;
      document.getElementById('flappyScore').textContent=flappyScore;
      if(flappyScore>flappyBest){flappyBest=flappyScore;document.getElementById('flappyBest').textContent=flappyBest;}
    }
  });
  // Collision
  if(bird.y+bird.radius>FH||bird.y-bird.radius<0){flappyRunning=false;flappyState='over';return;}
  pipes.forEach(p=>{
    if(bird.x+bird.radius-8>p.x&&bird.x-bird.radius+8<p.x+PIPE_W){
      if(bird.y-bird.radius<p.top||bird.y+bird.radius>p.top+GAP){
        flappyRunning=false;flappyState='over';
      }
    }
  });
}

function drawFlappy(){
  // Sky gradient
  const sky=fx.createLinearGradient(0,0,0,FH);
  sky.addColorStop(0,'#0a0a2e');sky.addColorStop(1,'#0d1b4b');
  fx.fillStyle=sky;fx.fillRect(0,0,FW,FH);
  // Stars
  fx.fillStyle='rgba(255,255,255,0.5)';
  for(let i=0;i<30;i++){fx.fillRect((i*73+flappyScore*0.3)%FW,(i*37)%200,1,1);}
  // Ground
  fx.fillStyle='#1a3a1a';fx.fillRect(0,FH-30,FW,30);
  fx.fillStyle='#2d5a1b';fx.fillRect(0,FH-34,FW,6);

  // Pipes
  pipes.forEach(p=>{
    // Top pipe
    const tg=fx.createLinearGradient(p.x,0,p.x+PIPE_W,0);
    tg.addColorStop(0,'#1a6b1a');tg.addColorStop(0.5,'#2d9e2d');tg.addColorStop(1,'#1a6b1a');
    fx.fillStyle=tg;fx.fillRect(p.x,0,PIPE_W,p.top);
    fx.fillStyle='#25851e';fx.fillRect(p.x-4,p.top-20,PIPE_W+8,20);
    // Bottom pipe
    fx.fillStyle=tg;fx.fillRect(p.x,p.top+GAP,PIPE_W,FH);
    fx.fillStyle='#25851e';fx.fillRect(p.x-4,p.top+GAP,PIPE_W+8,20);
    // Glow
    fx.shadowColor='#00ff44';fx.shadowBlur=8;
    fx.strokeStyle='rgba(0,255,68,0.3)';fx.lineWidth=1;
    fx.strokeRect(p.x,0,PIPE_W,p.top);
    fx.strokeRect(p.x,p.top+GAP,PIPE_W,FH);
    fx.shadowBlur=0;
  });

  // Bird
  fx.save();
  fx.translate(bird.x,bird.y);
  fx.rotate(bird.rot*Math.PI/180);
  // Body
  fx.fillStyle='#ff6b35';fx.shadowColor='#ff6b35';fx.shadowBlur=12;
  fx.beginPath();fx.ellipse(0,0,bird.radius,bird.radius-3,0,0,Math.PI*2);fx.fill();
  // Wing
  fx.fillStyle='#ff9a5c';
  fx.beginPath();fx.ellipse(-4,4,8,5,0.3,0,Math.PI*2);fx.fill();
  // Eye
  fx.fillStyle='#fff';fx.beginPath();fx.arc(6,-4,4,0,Math.PI*2);fx.fill();
  fx.fillStyle='#000';fx.beginPath();fx.arc(7,-4,2,0,Math.PI*2);fx.fill();
  // Beak
  fx.fillStyle='#ffd700';
  fx.beginPath();fx.moveTo(10,-2);fx.lineTo(18,0);fx.lineTo(10,3);fx.closePath();fx.fill();
  fx.shadowBlur=0;
  fx.restore();

  // Score
  fx.fillStyle='rgba(0,0,0,0.4)';fx.fillRect(FW/2-40,16,80,34);
  fx.fillStyle='#fff';fx.font='bold 20px "Press Start 2P"';fx.textAlign='center';
  fx.fillText(flappyScore,FW/2,42);
}

document.addEventListener('keydown', e => {
  // Snake
  if(activeGame==='snake'){
    const map={ArrowUp:[0,-1],ArrowDown:[0,1],ArrowLeft:[-1,0],ArrowRight:[1,0],w:[0,-1],s:[0,1],a:[-1,0],d:[1,0]};
    const d=map[e.key];
    if(d&&!(snakeDir[0]===-d[0]&&snakeDir[1]===-d[1])){snakeDir=d;e.preventDefault();}
  }
  // Pacman
  if(activeGame==='pacman'){
    const map={ArrowUp:[0,-1],ArrowDown:[0,1],ArrowLeft:[-1,0],ArrowRight:[1,0]};
    if(map[e.key]){pacNextDir=map[e.key];e.preventDefault();}
  }
  // Tetris
  if(activeGame==='tetris'){
    if(e.key==='ArrowLeft'){tetMoveLeft();e.preventDefault();}
    if(e.key==='ArrowRight'){tetMoveRight();e.preventDefault();}
    if(e.key==='ArrowDown'){tetMoveDown();e.preventDefault();}
    if(e.key==='ArrowUp'){tetRotate();e.preventDefault();}
    if(e.key===' '){tetHardDrop();e.preventDefault();}
  }
  // Flappy
  if(activeGame==='flappy'){
    if(e.key===' '||e.key==='ArrowUp'){flappyFlap();e.preventDefault();}
  }
});
